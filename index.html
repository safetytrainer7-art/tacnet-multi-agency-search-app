// Add Search Control (Geocoding) const searchControl = L.Control.geocoder({ defaultMarkGeocode: true, placeholder: "Enter Address..." }).addTo(map); // Add Map Layer Selection (Satellite, Street, Terrain) const baseMaps = { "Street": L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png'), "Satellite": L.tileLayer('https://server.arcgisonline.com/ArcGIS/rest/services/World_Imagery/MapServer/tile/{z}/{y}/{x}'), "Terrain": L.tileLayer('https://{s}.tile.opentopomap.org/{z}/{x}/{y}.png') }; L.control.layers(baseMaps).addTo(map); // Targeting Feature: Call this when address is found function placeTarget(latlng) { L.circle(latlng, { radius: 100, color: 'lime', fillColor: 'lime' }).addTo(map); L.marker(latlng).addTo(map).bindPopup("TARGET AREA"); }
